	int d[300], len;

sd read_num()
{
	sd x;
	memset(x.d, 0, sizeof(x.d));
	char s[300];
	cin >> s;
	x.len = strlen(s);
	int k = 0;
	for(int i=x.len-1; i>=0; --i)
		x.d[k++] = s[i]-'0';
	return x;
}
sd Plus(sd x, sd y)
{
	sd t;
	memset(t.d, 0, sizeof(t.d));
	int len = x.len>y.len?x.len:y.len;
	//printf("%d\n", len);
	for(int i=0; i<len; ++i)
	{
		//printf("%d %d\n", x.d[i], y.d[i]);
		t.d[i] += x.d[i]+y.d[i];
		//printf("%d ", (int)t.d[i]);
		t.d[i+1] = t.d[i]/10;
		//printf("%d ", (int)t.d[i+1]);
		t.d[i] = t.d[i]%10;
		//printf("%d\n", (int)t.d[i]);
	}
	if(t.d[len] > 0)
		t.len = len+1;
	else
		t.len = len;
	return t;
}
void print(sd x)
{
	for(int i=x.len-1; i>=0; --i)
		printf("%d", (int)x.d[i]);
	return;
}

{
	sd t;
	memset(t.d, 0, sizeof(t.d));
	//int len = x.len>y.len?x.len:y.len;
	//printf("%d\n", len);
	for(int i=0; i<x.len; ++i)
	{
		int temp = 0;

			t.d[i+j] = x.d[i]*y.d[j]+temp+t.d[i+j];

	return t;
}
