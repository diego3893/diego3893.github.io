void qsort(int left, int right)
{

		return;

	tmp = q2[left];
	int i=left, j=right;
	while(i != j)
	{
		while(q2[j]>=tmp && i<j)
			--j;
		while(q2[i]<=tmp && i<j)
			++i;

	q2[i] = tmp;
	qsort(left, i-1);
	qsort(i+1, right);
	return;
}
